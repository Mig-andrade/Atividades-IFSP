algoritmo "VerificaNumero"
var
   numero: inteiro
inicio
   escreva("Digite um número inteiro: ")
   leia(numero)

   se (numero > 0) entao
      escreval("O número é Positivo.")
   senao
      se (numero < 0) entao
         escreval("O número é Negativo.")
      senao
         escreval("O número é Zero.")
      fimse
   fimse
fimalgoritmo

algoritmo "CalculoMedia"
var
   nota1, nota2, media: real
inicio
   escreva("Digite a primeira nota: ")
   leia(nota1)
   escreva("Digite a segunda nota: ")
   leia(nota2)

   media <- (nota1 + nota2) / 2

   se (media >= 6) entao
      escreval("Média: ", media, " - Status: APROVADO")
   senao
      escreval("Média: ", media, " - Status: REPROVADO")
   fimse
fimalgoritmo

algoritmo "DescontoProduto"
var
   preco, total, valorFinal: real
   quantidade: inteiro
inicio
   escreva("Preço do produto: R$ ")
   leia(preco)
   escreva("Quantidade: ")
   leia(quantidade)

   total <- preco * quantidade

   se (total >= 100) entao
      valorFinal <- total * 0.90
      escreval("Total Bruto: R$ ", total)
      escreval("Valor com 10% de desconto: R$ ", valorFinal)
   senao
      escreval("Total a pagar: R$ ", total)
   fimse
fimalgoritmo

algoritmo "ContaPares"
var
   i, numero, contPares: inteiro
inicio
   contPares <- 0

   para i de 1 ate 10 faca
      escreva("Digite o ", i, "º número: ")
      leia(numero)
      
      se (numero % 2 = 0) entao
         contPares <- contPares + 1
      fimse
   fimpara

   escreval("Total de números pares digitados: ", contPares)
fimalgoritmo

Var
    num, soma: inteiro


Inicio
    soma <- 0;
    num  <- 1;

    Enquanto (num != 0)
    	Escreva("Digite um número (digite 0 para parar)")
    	Leia(num)
	soma <- soma + num
    FimEnquanto
	Escreva("A soma dos números digitados foi: ", soma)
